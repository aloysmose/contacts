namespace Contacts;

/**
 * ContactInterface for all `Contact` child classes
 */
interface ContactInterface
{
    public function addFullName(string $name): void;

    public function addName(string $lastName, ?string $firstName, ?string $additionalName, ?string $prefix, ?string $suffix): void;

    public function addNicknames(array $names): void;

    public function addPhoto(string $photo, ?bool $isUrl): void;

    public function addBirthday(?int $year, int $month, int $day): void;

    public function addAddress(?string $poBox, ?string $extended, ?string $street, ?string $city, ?string $state, ?string $zip, ?string $country, ?array $types): void;

    public function addLabel(string $label, ?array $types): void;

    public function addTelephone(string $phone, ?array $types): void;

    public function addEmail(string $email, ?array $types): void;

    public function addMailer(string $mailer): void;

    public function addTimeZone(string $timeZone): void;

    public function addLatLong(string $lat, string $long): void;

    public function addTitle(string $title): void;

    public function addRole(string $role): void;

    public function addLogo(string $logo): void;

    public function addAgent(?string $agent): void;

    public function addOrganizations(array $organizations): void;

    public function addCategories(array $categories): void;

    public function addNote(string $note): void;

    public function addProductId(string $productId): void;

    public function addRevision(): void;

    public function addSortString(string $sortString): void;

    public function addSound(?string $sound): void;

    public function addUniqueIdentifier(string $uniqueIdentifier): void;

    public function addUrl(string $url): void;

    public function addClassification(?string $classification): void;

    public function addKey(?string $key): void;

    public function addExtendedType(string $label, string $value): void;
}

?>
